231 void MainController::exhale() {
232  // Control loop
233  m_ventilationController->exhale();

235  // Compute the PEEP pressure
236  int16_t minValue = m_lastPressureValues[0u];

257 void MainController::endRespiratoryCycle(uint32_t p_currentMillis) {
258  // Compute the respiratory rate: average on NUMBER_OF_BREATH_PERIOD breaths
259  uint32_t currentMillis = p_currentMillis;
260  m_lastBreathPeriodsMs[m_lastBreathPeriodsMsIndex] =
261  currentMillis - m_lastEndOfRespirationDateMs;
262  m_lastBreathPeriodsMsIndex++;
263  if (m_lastBreathPeriodsMsIndex >= NUMBER_OF_BREATH_PERIOD) {
264  m_lastBreathPeriodsMsIndex = 0;
265  }
266  uint32_t sum = 0;
267  for (uint8_t i = 0u; i < NUMBER_OF_BREATH_PERIOD; i++) {
268  sum += m_lastBreathPeriodsMs[i];
269  }
270  // Add "+(sum-1u)" to round instead of truncate
271  m_cyclesPerMinuteMeasure = (((NUMBER_OF_BREATH_PERIOD * 60u) * 1000u) + (sum - 1u)) / sum;
272  m_lastEndOfRespirationDateMs = currentMillis;
273 
274  // Plateau pressure is the mean pressure during plateau
275  m_plateauPressureMeasure = max(0, static_cast<int16_t>(m_PlateauMeasureSum)
276  / static_cast<int16_t>(m_PlateauMeasureCount));
277 
278  checkCycleAlarm();
279 
280  m_plateauPressureToDisplay = m_plateauPressureMeasure;
281 
282  // Send telemetry machine state snapshot message
283  sendMachineState();
284 
285  m_ventilationController->endCycle();
286 }

363 void MainController::computeTickParameters() {
364  // Inspiratory term is always 10. Expiratory term is between 10 and 60 (default 20).
365  // The folowing calculation is equivalent of 1000 * (10 / (10 + m_expiratoryTerm) * (60 /
366  // m_cyclesPerMinute).
367  m_plateauDurationMs =
368  ((10000u / (10u + m_expiratoryTermCommand)) * 60u) / m_cyclesPerMinuteCommand;
369 
370  m_ticksPerCycle =
371  60u * (1000000u / MAIN_CONTROLLER_COMPUTE_PERIOD_MICROSECONDS) / m_cyclesPerMinuteCommand;
372  m_ticksPerInhalation =
373  (m_plateauDurationMs * 1000000u / MAIN_CONTROLLER_COMPUTE_PERIOD_MICROSECONDS) / 1000u;
