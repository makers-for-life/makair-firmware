155 #define SM_DEBOUNCE 2 // number of debounce ticks before trigger an action

159 // fast solution : no hardware timer, no interrupt priority or atomicity issue. Close to Arduino
160 // philosophy.
161 void scanMatrixLoop() {
162  if (1 == scanMatrixCurrentColumn) {
163  // Increase counter for each column and row
164  if (HIGH == digitalRead(PIN_IN_ROW1)) {
165  scanMatrixCounterC1R1++;
166  } else {
167  scanMatrixCounterC1R1 = 0;
168  }
169  if (HIGH == digitalRead(PIN_IN_ROW2)) {
170  scanMatrixCounterC1R2++;
171  } else {
172  scanMatrixCounterC1R2 = 0;
173  }
174  if (HIGH == digitalRead(PIN_IN_ROW3)) {
175  scanMatrixCounterC1R3++;
176  } else {
177  scanMatrixCounterC1R3 = 0;
178  }
179  // first click (after debounce ticks) or
180  // later clicks if continuous press trigger action
181  if (SM_DEBOUNCE == scanMatrixCounterC1R1
182  || (scanMatrixCounterC1R1 >= SM_REPEAT
183  && 0 == (scanMatrixCounterC1R1 - SM_REPEAT) % SM_PERIOD)) {
184  onPeakPressureIncrease();
185  }
186  if (SM_DEBOUNCE == scanMatrixCounterC1R2
187  || (scanMatrixCounterC1R2 >= SM_REPEAT
188  && 0 == (scanMatrixCounterC1R2 - SM_REPEAT) % SM_PERIOD)) {
189  onPeakPressureDecrease();
190  }
191  if (SM_DEBOUNCE == scanMatrixCounterC1R3
192  || (scanMatrixCounterC1R3 >= SM_REPEAT
193  && 0 == (scanMatrixCounterC1R3 - SM_REPEAT) % SM_PERIOD)) {
194  onPlateauPressureIncrease();
195  }
196  } else if (2 == scanMatrixCurrentColumn) {
197  if (HIGH == digitalRead(PIN_IN_ROW1)) {
198  scanMatrixCounterC2R1++;
199  } else {
200  scanMatrixCounterC2R1 = 0;
201  }
202  if (HIGH == digitalRead(PIN_IN_ROW2)) {
203  scanMatrixCounterC2R2++;
204  } else {
205  scanMatrixCounterC2R2 = 0;
206  }
207  if (HIGH == digitalRead(PIN_IN_ROW3)) {
208  scanMatrixCounterC2R3++;
209  } else {
210  scanMatrixCounterC2R3 = 0;
211  }
212  // first click (after debounce ticks) or
213  // later clicks if continuous press trigger action
214  if (SM_DEBOUNCE == scanMatrixCounterC2R1
215  || (scanMatrixCounterC2R1 >= SM_REPEAT
216  && 0 == (scanMatrixCounterC2R1 - SM_REPEAT) % SM_PERIOD)) {
217  onPlateauPressureDecrease();
218  }
219  if (SM_DEBOUNCE == scanMatrixCounterC2R2
220  || (scanMatrixCounterC2R2 >= SM_REPEAT
221  && 0 == (scanMatrixCounterC2R2 - SM_REPEAT) % SM_PERIOD)) {
222  onPeepPressureIncrease();
223  }
224  if (SM_DEBOUNCE == scanMatrixCounterC2R3
225  || (scanMatrixCounterC2R3 >= SM_REPEAT
226  && 0 == (scanMatrixCounterC2R3 - SM_REPEAT) % SM_PERIOD)) {
227  onPeepPressureDecrease();
228  }
229  } else if (3 == scanMatrixCurrentColumn) {
230  if (HIGH == digitalRead(PIN_IN_ROW1)) {
231  scanMatrixCounterC3R1++;
232  } else {
233  scanMatrixCounterC3R1 = 0;
234  }

241  scanMatrixCounterC3R3++;

244  }
245  // first click (after debounce ticks) or

247  if (SM_DEBOUNCE == scanMatrixCounterC3R1
248  || (scanMatrixCounterC3R1 >= SM_REPEAT
249  && 0 == (scanMatrixCounterC3R1 - SM_REPEAT) % SM_PERIOD)) {
250  onCycleIncrease();
251  }
252  if (SM_DEBOUNCE == scanMatrixCounterC3R2
253  || (scanMatrixCounterC3R2 >= SM_REPEAT
254  && 0 == (scanMatrixCounterC3R2 - SM_REPEAT) % SM_PERIOD)) {
255  onCycleDecrease();
256  }
257  // there is no button on col3 x row3
258  }
259 
260  // next column
261  scanMatrixCurrentColumn++;
262  if (4 == scanMatrixCurrentColumn) {
263  scanMatrixCurrentColumn = 1;
264  }
265  digitalWrite(PIN_OUT_COL1, 1 == scanMatrixCurrentColumn ? HIGH : LOW);
266  digitalWrite(PIN_OUT_COL2, 2 == scanMatrixCurrentColumn ? HIGH : LOW);
267  digitalWrite(PIN_OUT_COL3, 3 == scanMatrixCurrentColumn ? HIGH : LOW);
268 }
