554 int8_t MFM_calibrateZero(void) {
555  int8_t ret = MFM_CALIBRATION_OK;
556  // activate table fill with last valid value
557  mfmInspiratoryInstantAirFlowLastValuesIndex = 0;
558  mfmInspiratoryInstantAirFlowRecord = true;
559  // wait for the table to fill in
560  delay(2 + (MFM_MEAN_SAMPLES * (MASS_FLOW_PERIOD / 10)));
561  // Check that table is full (record must be false)
562  // If it is not, there is a sensor problem
563  // In case of problem, do not update mfmInspiratoryCalibrationOffset
564  if (!mfmInspiratoryInstantAirFlowRecord) {
565  int32_t zeroFlow = 0;
566  for (int16_t i = 0; i < MFM_MEAN_SAMPLES; i++) {
567  zeroFlow += mfmInspiratoryInstantAirFlowLastValues[i];
568  }
569  zeroFlow /= MFM_MEAN_SAMPLES;
570  // check that value is credible: [-10 10] SLPM
571  if ((zeroFlow < 10000) && (zeroFlow > -10000)) {
572  mfmInspiratoryCalibrationOffset = zeroFlow;
573  } else {
574  ret = MFM_CALIBRATION_OUT_OF_RANGE;
575  }
576  } else {
577  ret = MFM_CALIBRATION_IMPOSSIBLE;
578  }
579  return ret;
580 }
