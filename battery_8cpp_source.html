42 void updateBatterySample() {

44 
45  // Substract previous sample
46  batteryTotalSamples = batteryTotalSamples - rawBatterySample[batteryCurrentSample];
47 
48  // Assign sample from Vout
49  rawBatterySample[batteryCurrentSample] = rawVout;
50

53

56 
57  // If we reached max samples

61
