47 void updateBatterySample() {
48  uint16_t rawVout = analogRead(PIN_BATTERY);
49 
50  // Assign sample from Vout
51  rawBatterySample[batteryCurrentSample] = rawVout;
52 
53  // Increment sample
54  batteryCurrentSample++;
55 
56  // If we reached max samples
57  if (batteryCurrentSample >= BATTERY_MAX_SAMPLES) {
58  batteryCurrentSample = 0;
59  }
60 
61  batteryTotalSamples = 0;
62  for (uint8_t i = 0; i < BATTERY_MAX_SAMPLES; i++) {
63  batteryTotalSamples += rawBatterySample[i];
64  }
65  // Updates mean voltage
66  rawBatteryMeanVoltage = (batteryTotalSamples / BATTERY_MAX_SAMPLES);
67 }
