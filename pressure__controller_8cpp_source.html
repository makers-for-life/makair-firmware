350  updatePhase(p_tick);

352 
353  // Compute metrics for alarms
354  m_sumOfPressures += m_pressure;
355  m_numberOfPressures++;
356 
357  // Act accordingly
358  switch (m_subPhase) {
359  case CycleSubPhases::INSPIRATION: {
360  inhale();

362  break;
363  }
364  case CycleSubPhases::HOLD_INSPIRATION: {

366  m_inhalationLastPressure = m_pressure;
367  break;
368  }

373  // plateau is computed differently
374 #if VALVE_TYPE != VT_FAULHABER
375  computePlateau(p_tick);
376 #endif
377  break;
378  }
379  }
380 
381  // RCM-SW-18
382  if (m_pressure > ALARM_THRESHOLD_MAX_PRESSURE) {
383  m_alarmController->detectedAlarm(RCM_SW_18, m_cycleNb, ALARM_THRESHOLD_MAX_PRESSURE,
384  m_pressure);
385  }

401 void PressureController::computePlateau(uint16_t p_tick) {
402  uint16_t minValue = m_lastPressureValues[0u];
403  uint16_t maxValue = m_lastPressureValues[0u];
404  uint16_t totalValues = m_lastPressureValues[0u];
405 
406  for (uint8_t index = 1u; index < MAX_PRESSURE_SAMPLES; index++) {
407  minValue = min(minValue, m_lastPressureValues[index]);
408  maxValue = max(maxValue, m_lastPressureValues[index]);
409  totalValues += m_lastPressureValues[index];
410  }
411 
412  uint16_t diff = (maxValue - minValue);
413 
414  // Start computing plateau pressure when:
415  // - the last pressure values were close enough
416  // - the hold inspiration phase is about to end
417  // - plateau pressure computation was not already started
418  if (!m_plateauComputed && (diff < 10u) && (p_tick >= ((m_tickPerInhalation * 95u) / 100u))) {
419  m_startPlateauComputation = true;
420  }
421 
422  // Stop computing plateau pressure when pressure drops
423  if (m_startPlateauComputation && (diff > 10u)) {
424  m_startPlateauComputation = false;
425  m_plateauComputed = true;
426  }
427 
428  if (m_startPlateauComputation) {
429  m_plateauPressure = totalValues / MAX_PRESSURE_SAMPLES;
430  }

974 int32_t PressureController::pidBlower(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
975 #if VALVE_TYPE == VT_FAULHABER
976 
977  int32_t minAperture = m_blower_valve.minAperture();
978  int32_t maxAperture = m_blower_valve.maxAperture();
979  uint32_t blowerAperture;
980  int32_t derivative = 0;
981  int32_t smoothError = 0;
982  int32_t totalValues = 0;
983  int32_t proportionnalWeight;
984  int32_t derivativeWeight;
985 
986  int32_t coefficientP;
987  int32_t coefficientI;
988  int32_t coefficientD;
989 
990  int32_t temporaryBlowerIntegral = 0;
991 
992  // Compute error
993  int32_t error = targetPressure - currentPressure;
994 
995  // Windowing. It overides the parameter.h coefficients
996  if (error < 0) {
997  coefficientI = 200;
998  coefficientP = 2500;
999  coefficientD = 0;
1000  } else {
1001  coefficientI = 50;
1002  coefficientP = 2500;
1003  coefficientD = 0;
1004  }
1005 
1006  // Calculate Derivative part. Include a moving average on error for smoothing purpose
1007  m_lastBlowerPIDError[m_lastBlowerPIDErrorIndex] = error;
1008  m_lastBlowerPIDErrorIndex++;
1009  if (m_lastBlowerPIDErrorIndex
1010  >= static_cast<int32_t>(NUMBER_OF_SAMPLE_BLOWER_DERIVATIVE_MOVING_MEAN)) {
1011  m_lastBlowerPIDErrorIndex = 0;
1012  }
1013  for (uint8_t i = 0u; i < NUMBER_OF_SAMPLE_BLOWER_DERIVATIVE_MOVING_MEAN; i++) {
1014  totalValues += m_lastBlowerPIDError[i];
1015  }
1016  smoothError =
1017  totalValues / static_cast<int32_t>(NUMBER_OF_SAMPLE_BLOWER_DERIVATIVE_MOVING_MEAN);
1018 
1019  // Fast mode ends at 20mmH20 from target. When changing from fast-mode to PID, Set the integral
1020  // to the previous value
1021  if (error < 20) {
1022  if (blowerPIDFastMode) {
1023  proportionnalWeight = (coefficientP * error) / 1000;
1024  derivativeWeight = (coefficientD * derivative / 1000);
1025  blowerIntegral =
1026  1000 * ((int32_t)lastBlowerAperture - maxAperture) / (minAperture - maxAperture)
1027  - (proportionnalWeight + derivativeWeight);
1028  }
1029  blowerPIDFastMode = false;
1030  }
1031 
1032  // In fast mode : everything is openned (Open loop)
1033  if (blowerPIDFastMode) {
1034  // Ramp from 125 to 0 angle during 250ms.
1035  int32_t increment = 5 * ((int32_t)PCONTROLLER_COMPUTE_PERIOD_US) / 10000;
1036  if (lastBlowerAperture >= static_cast<uint32_t>(abs(increment))) {
1037  blowerAperture =
1038  max(minAperture,
1039  min(maxAperture, (static_cast<int32_t>(lastBlowerAperture) - increment)));
1040  } else {
1041  blowerAperture = 0;
1042  }
1043  } else { // Then use the PID to finish slowly
1044  derivative = ((dt == 0)) ? 0 : ((1000000 * (blowerLastError - smoothError)) / dt);
1045 
1046  temporaryBlowerIntegral = blowerIntegral + ((coefficientI * error * dt) / 1000000);
1047  temporaryBlowerIntegral =
1048  max(PID_BLOWER_INTEGRAL_MIN, min(PID_BLOWER_INTEGRAL_MAX, temporaryBlowerIntegral));
1049 
1050  proportionnalWeight = ((coefficientP * error) / 1000);
1051  int32_t integralWeight = temporaryBlowerIntegral;
1052  derivativeWeight = coefficientD * derivative / 1000;
1053 
1054  int32_t blowerCommand = proportionnalWeight + integralWeight + derivativeWeight;
1055  blowerAperture =
1056  max(minAperture,
1057  min(maxAperture, maxAperture + (minAperture - maxAperture) * blowerCommand / 1000));
1058  }
1059 
1060  // If the valve is completly open or completly closed, dont update Integral
1061  if ((blowerAperture != static_cast<uint32_t>(minAperture))
1062  && (blowerAperture != static_cast<uint32_t>(maxAperture))) {
1063  blowerIntegral = temporaryBlowerIntegral;
1064  }
1065 
1066  lastBlowerAperture = blowerAperture;
1067  blowerLastError = smoothError;
1068 
1069 #else
1070  // Compute error
1071  int32_t error = targetPressure - currentPressure;
1072 
1073  // Compute integral
1074  blowerIntegral = blowerIntegral + ((PID_BLOWER_KI * error * dt) / 1000000);
1075  blowerIntegral = max(PID_BLOWER_INTEGRAL_MIN, min(PID_BLOWER_INTEGRAL_MAX, blowerIntegral));
1076 
1077  // Compute derivative
1078  int32_t derivative = ((blowerLastError == INVALID_ERROR_MARKER) || (dt == 0))
1079  ? 0
1080  : ((1000000 * (error - blowerLastError)) / dt);
1081  blowerLastError = error;
1082 
1083  int32_t blowerCommand = (PID_BLOWER_KP * error) + blowerIntegral
1084  + ((PID_BLOWER_KD * derivative) / 1000); // Command computation
1085 
1086  int32_t minAperture = m_blower_valve.minAperture();
1087  int32_t maxAperture = m_blower_valve.maxAperture();
1088 
1089  uint32_t blowerAperture =
1090  max(minAperture,
1091  min(maxAperture, maxAperture + (minAperture - maxAperture) * blowerCommand / 1000));
1092 #endif
1093 
1094  return blowerAperture;
1095 }
