132 void PC_CMV_Controller::calculateBlowerIncrement() {
133  int16_t peakDelta =
134  mainController.peakPressureMeasure() - mainController.plateauPressureCommand();
135 
136  // Number of tick for the half ramp (120 ms)
137  int32_t halfRampNumberfTick =
138  1000 * 120 / static_cast<int32_t>(MAIN_CONTROLLER_COMPUTE_PERIOD_MICROSECONDS);
139 
140  // Update blower only if patient is plugged on the machine
141  if (mainController.peakPressureMeasure() > 20) {
142  if (m_plateauStartTime < ((mainController.ticksPerInhalation() * 30u) / 100u)) {
143  // Only case for decreasing the blower: ramping is too fast or overshooting is too high
144  if ((m_plateauStartTime < static_cast<uint32_t>(abs(halfRampNumberfTick)))
145  || ((peakDelta > 15)
146  && (m_plateauStartTime < ((mainController.ticksPerInhalation() * 20u) / 100u)))
147  || (peakDelta > 25)) {
148  m_blowerIncrement = -100;
149  DBG_DO(Serial.println("BLOWER -100");)

151  m_blowerIncrement = 0;
152  DBG_DO(Serial.println("BLOWER 0");)
153  }

155  DBG_DO(Serial.println("BLOWER +0");)
156  m_blowerIncrement = 0;
157  } else if (m_plateauStartTime < ((mainController.ticksPerInhalation() * 50u) / 100u)) {

174 PC_CMV_Controller::PCinspiratoryPID(int32_t targetPressure, int32_t currentPressure, int32_t dt) {
175  int32_t minAperture = inspiratoryValve.minAperture();
176  int32_t maxAperture = inspiratoryValve.maxAperture();
177  int32_t inspiratoryValveAperture;
178  int32_t derivative = 0;
179  int32_t smoothError = 0;
180  int32_t totalValues = 0;
181  int32_t proportionnalWeight;
182  int32_t derivativeWeight;
183 
184  int32_t coefficientP;
185  int32_t coefficientI;
186  int32_t coefficientD;
187 
188  int32_t temporarym_inspiratoryPidIntegral = 0;
189 
190  // Compute error
191  int32_t error = targetPressure - currentPressure;
192 
193  // Windowing (it overrides the parameter.h coefficients)
194  if (error < 0) {
195  coefficientI = 200;
196  coefficientP = 2500;
197  coefficientD = 0;
198  } else {
199  coefficientI = 50;
200  coefficientP = 2500;
201  coefficientD = 0;
202  }
203 
204  // Calculate the derivative part
205  // Include a moving average on error for smoothing purpose
206  m_inspiratoryPidLastErrors[m_inspiratoryPidLastErrorsIndex] = error;
207  m_inspiratoryPidLastErrorsIndex++;
208  if (m_inspiratoryPidLastErrorsIndex
209  >= static_cast<int32_t>(PC_NUMBER_OF_SAMPLE_DERIVATIVE_MOVING_MEAN)) {
210  m_inspiratoryPidLastErrorsIndex = 0;
211  }
212  for (uint8_t i = 0u; i < PC_NUMBER_OF_SAMPLE_DERIVATIVE_MOVING_MEAN; i++) {
213  totalValues += m_inspiratoryPidLastErrors[i];
214  }
215  smoothError = totalValues / static_cast<int32_t>(PC_NUMBER_OF_SAMPLE_DERIVATIVE_MOVING_MEAN);
216 
217  // Fast mode ends at 20 mmH20 from target
218  // When changing from fast mode to PID, set the integral to the previous value
219  if (error < 20) {
220  if (m_inspiratoryPidFastMode) {
221  proportionnalWeight = (coefficientP * error) / 1000;
222  derivativeWeight = (coefficientD * derivative / 1000);
223  m_inspiratoryPidIntegral = 1000
224  * ((int32_t)m_inspiratoryValveLastAperture - maxAperture)
225  / (minAperture - maxAperture)
226  - (proportionnalWeight + derivativeWeight);
227  }
228  m_inspiratoryPidFastMode = false;
229  }
230 
231  // In fast mode: everything is openned (open loop)
232  if (m_inspiratoryPidFastMode) {
233  // Ramp from 125 to 0 angle during 250 ms
234  int32_t increment =
235  (5 * static_cast<int32_t>(MAIN_CONTROLLER_COMPUTE_PERIOD_MICROSECONDS)) / 10000;
236  if (m_inspiratoryValveLastAperture >= abs(increment)) {
237  inspiratoryValveAperture =
238  max(minAperture, min(maxAperture, m_inspiratoryValveLastAperture - increment));
239  } else {
240  inspiratoryValveAperture = 0;
241  }
242  } else { // If not in fast mode, the PID is used
243  derivative = ((dt == 0)) ? 0 : ((1000000 * (m_inspiratoryPidLastError - smoothError)) / dt);
244 
245  temporarym_inspiratoryPidIntegral =
246  m_inspiratoryPidIntegral + ((coefficientI * error * dt) / 1000000);
247  temporarym_inspiratoryPidIntegral =
248  max(PID_BLOWER_INTEGRAL_MIN,
249  min(PID_BLOWER_INTEGRAL_MAX, temporarym_inspiratoryPidIntegral));
250 
251  proportionnalWeight = ((coefficientP * error) / 1000);
252  int32_t integralWeight = temporarym_inspiratoryPidIntegral;
253  derivativeWeight = coefficientD * derivative / 1000;
254 
255  int32_t blowerCommand = proportionnalWeight + integralWeight + derivativeWeight;
256  inspiratoryValveAperture =
257  max(minAperture,
258  min(maxAperture, maxAperture + (minAperture - maxAperture) * blowerCommand / 1000));
259  }
260 
261  // If the valve is completely open or completely closed, don't update the integral part
262  if ((inspiratoryValveAperture != minAperture) && (inspiratoryValveAperture != maxAperture)) {
263  m_inspiratoryPidIntegral = temporarym_inspiratoryPidIntegral;
264  }
265 
266  m_inspiratoryValveLastAperture = inspiratoryValveAperture;
267  m_inspiratoryPidLastError = smoothError;
268 
269  return inspiratoryValveAperture;
270 }

275  int32_t maxAperture = expiratoryValve.maxAperture();
276  int32_t expiratoryValveAperture;
277  int32_t derivative = 0;
278  int32_t smoothError = 0;
279  int32_t totalValues = 0;
280  int32_t temporarym_expiratoryPidIntegral = 0;
281  int32_t proportionnalWeight;
282  int32_t derivativeWeight;
283 
284  int32_t coefficientP;
285  int32_t coefficientI;
286  int32_t coefficientD;
287 
288  // Compute error
289  int32_t error = targetPressure + PID_PATIENT_SAFETY_PEEP_OFFSET - currentPressure;
290 
291  // Calculate derivative part
292  // Include a moving average on error for smoothing purpose
293  m_expiratoryPidLastErrors[m_expiratoryPidLastErrorsIndex] = error;
294  m_expiratoryPidLastErrorsIndex++;
295  if (m_expiratoryPidLastErrorsIndex
296  >= static_cast<int32_t>(PC_NUMBER_OF_SAMPLE_DERIVATIVE_MOVING_MEAN)) {
297  m_expiratoryPidLastErrorsIndex = 0;
298  }
299  for (uint8_t i = 0u; i < PC_NUMBER_OF_SAMPLE_DERIVATIVE_MOVING_MEAN; i++) {
300  totalValues += m_expiratoryPidLastErrors[i];
301  }
302  smoothError = totalValues / static_cast<int32_t>(PC_NUMBER_OF_SAMPLE_DERIVATIVE_MOVING_MEAN);
303  derivative = (dt == 0) ? 0 : (1000000 * (smoothError - m_expiratoryPidLastError)) / dt;
304 
305  // Windowing (it overrides the parameter.h coefficients)
306  if (error < 0) {
307  coefficientI = 50;
308  coefficientP = 2500;
309  coefficientD = 0;
310  } else {
311  // For a high PEEP, a lower KI is required
312  // For PEEP = 100 mmH2O, KI = 120
313  // For PEEP = 50 mmH2O, KI = 250
314  if (mainController.peepCommand() > 100) {
315  coefficientI = 120;
316  } else {
317  coefficientI = ((-130 * ((int32_t)mainController.peepCommand())) / 50) + 380;
318  }
319 
320  coefficientP = 2500;
321  coefficientD = 0;
322  }
323 
324  // Fast mode ends at 30 mmH20 from target
325  // When changing from fast mode to PID, set the integral to the previous value
326  if (error > -30) {
327  if (m_expiratoryPidFastMode) {
328  proportionnalWeight = (coefficientP * error) / 1000;
329  derivativeWeight = (coefficientD * derivative / 1000);
330  m_expiratoryPidIntegral = 1000 * ((int32_t)m_expiratoryValveLastAperture - maxAperture)
331  / (maxAperture - minAperture)
332  - (proportionnalWeight + derivativeWeight);
333  }
334  m_expiratoryPidFastMode = false;
335  }
336 
337  // Fast mode: open loop with ramp
338  if (m_expiratoryPidFastMode) {
339  // Ramp from 125 to 0 angle during 250 ms
340  int32_t increment =
341  (5 * static_cast<int32_t>(MAIN_CONTROLLER_COMPUTE_PERIOD_MICROSECONDS)) / 10000;
342  if (m_expiratoryValveLastAperture >= abs(increment)) {
343  expiratoryValveAperture =
344  max(minAperture, min(maxAperture, m_expiratoryValveLastAperture - increment));
345  } else {
346  expiratoryValveAperture = 0;
347  }
348  } else { // If not in fast mode, the PID is used
349  temporarym_expiratoryPidIntegral =
350  m_expiratoryPidIntegral + ((coefficientI * error * dt) / 1000000);
351  temporarym_expiratoryPidIntegral =
352  max(PID_PATIENT_INTEGRAL_MIN,
353  min(PID_PATIENT_INTEGRAL_MAX, temporarym_expiratoryPidIntegral));
354 
355  proportionnalWeight = ((coefficientP * error) / 1000);
356  int32_t integralWeight = temporarym_expiratoryPidIntegral;
357  derivativeWeight = coefficientD * derivative / 1000;
358 
359  int32_t patientCommand = proportionnalWeight + integralWeight + derivativeWeight;
360 
361  expiratoryValveAperture = max(
362  minAperture,
363  min(maxAperture, maxAperture + (maxAperture - minAperture) * patientCommand / 1000));
364  }
365 
366  // If the valve is completely open or completely closed, don't update integral part
367  if ((expiratoryValveAperture != minAperture) && (expiratoryValveAperture != maxAperture)) {
368  m_expiratoryPidIntegral = temporarym_expiratoryPidIntegral;
369  }
370 
371  m_expiratoryPidLastError = smoothError;
372  m_expiratoryValveLastAperture = expiratoryValveAperture;
373 
374  return expiratoryValveAperture;
